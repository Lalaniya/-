一、内存管理与垃圾回收 ​

1. 简述 JavaScript 的垃圾回收机制 ​

JavaScript 的垃圾回收机制是自动内存管理系统，通过识别和回收不再使用的内存空间来优化内存使用。主要算法有：

标记清除（Mark-Sweep）：

  1. 标记所有可达对象（从根对象开始遍历）
  2. 清除未标记对象
  3. 移动对象以整理内存碎片

标记整理（Mark-Compact）：
类似标记清除，但会移动对象以消除碎片。

分代回收（Generational GC）：
将对象分为新生代（短期对象）和老生代（长期对象），分别采用不同的回收策略。

代码示例：

javascript
// 闭包导致的内存泄漏示例
function createLeak() {
  const largeArray = new Array(1000).fill("data");
  return function () {
    return largeArray; // 闭包持有对 largeArray 的引用
  };
}

const leakyFunc = createLeak(); // largeArray 无法被回收

2. 什么是内存泄漏？如何检测和避免？ ​

内存泄漏指程序中已不再使用的内存无法被垃圾回收。常见原因：

  1. 全局变量：意外创建的全局变量
  2. 未清除的定时器/回调：setInterval未清除
  3. 闭包：闭包持有对外部变量的引用
  4. DOM 引用：DOM 元素已移除但 JavaScript 仍保留引用

检测方法：

  • Chrome DevTools 的 Memory 面板
  • Lighthouse 性能审计
  • 使用window.performance.memory监控内存使用

避免策略：

javascript
// 错误示例：全局变量泄漏
function leakExample() {
  leakedVar = "Oops!"; // 未声明的变量成为全局变量
}

// 正确做法：使用局部变量
function safeExample() {
  const localVar = "Safe"; // 使用 const/let
}

// 定时器泄漏
const interval = setInterval(() => {
  console.log("Running...");
}, 1000);

// 清除定时器
clearInterval(interval);

4. 什么是弱引用（Weak Reference）？如何使用？ ​

弱引用是一种不会阻止对象被垃圾回收的引用。JavaScript 中通过WeakMap和WeakSet实现。

主要特点：

  • 键必须是对象（WeakMap）
  • 无法枚举或遍历
  • 当对象的其他引用被释放时，WeakMap 中的引用不会阻止 GC

使用场景：

  • 缓存（不阻止对象被回收）
  • DOM 元素元数据存储
  • 避免内存泄漏

代码示例：

javascript
// WeakMap 缓存 DOM 元素的元数据
const elementMetadata = new WeakMap();

function setElementData(element, data) {
  elementMetadata.set(element, data);
}

function getElementData(element) {
  return elementMetadata.get(element);
}

// 当 DOM 元素被移除时，相关元数据会自动被 GC 回收

5. 什么是分代垃圾回收？它的优势是什么？ ​

分代垃圾回收基于"大多数对象生命周期很短"的观察，将内存分为：

  1. 新生代（Young Generation）：

    • 新创建的对象
    • 快速回收（Minor GC）
    • 使用 Scavenge 算法（复制存活对象）

  2. 老生代（Old Generation）：

    • 长期存活的对象（经历多次 Minor GC）
    • 较慢的回收（Major GC）
    • 使用标记清除或标记整理算法

优势：

  • 提高回收效率：大多数垃圾在新生代被回收
  • 减少停顿时间：分代处理降低单次 GC 的影响
  • 优化内存碎片：老生代采用整理算法

6. 实现一个带并发限制的异步调度器 ​

代码示例：

javascript
class Scheduler {
  constructor(maxConcurrency) {
    this.maxConcurrency = maxConcurrency;
    this.runningCount = 0;
    this.taskQueue = [];
  }

  async addTask(task) {
    if (this.runningCount >= this.maxConcurrency) {
      await new Promise((resolve) => this.taskQueue.push(resolve));
    }

    this.runningCount++;
    try {
      return await task();
    } finally {
      this.runningCount--;
      if (this.taskQueue.length > 0) {
        this.taskQueue.shift()();
      }
    }
  }
}

// 使用示例
const scheduler = new Scheduler(2);

async function task(id) {
  console.log(`Task ${id} started`);
  await new Promise((resolve) => setTimeout(resolve, 1000));
  console.log(`Task ${id} completed`);
  return id;
}

// 添加 5 个任务，并发限制为 2
Promise.all([
  scheduler.addTask(() => task(1)),
  scheduler.addTask(() => task(2)),
  scheduler.addTask(() => task(3)),
  scheduler.addTask(() => task(4)),
  scheduler.addTask(() => task(5)),
]).then((results) => console.log("All tasks completed:", results));

7. 解释 Event Loop 与任务队列的关系 ​

JavaScript 的执行机制基于事件循环（Event Loop）和任务队列（Task Queue）：

  1. 调用栈（Call Stack）：执行同步代码
  2. 任务队列：存放异步任务的回调
    • 宏任务（MacroTask）：setTimeout, setInterval, I/O, UI 渲染
    • 微任务（MicroTask）：Promise.then, MutationObserver, process.nextTick

事件循环工作流程：

  1. 执行完调用栈中的所有同步代码
  2. 处理微任务队列中的所有任务（直到队列为空）
  3. 执行一个宏任务
  4. 重复步骤 2 和 3

  console.log("Timeout");
}, 0);

});

// 输出顺序：Start -> End -> Promise -> Timeout

14. 如何实现一个记忆化（Memoization）的斐波那契函数？ ​

javascript
function memoize(func) {
  const cache = new Map();
  return function (...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      return cache.get(key);
    }
    const result = func.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// 斐波那契函数
const fibonacci = memoize(function (n) {
  if (n <= 1) {
    return n;
  }
  return fibonacci(n - 1) + fibonacci(n - 2);
});

console.log(fibonacci(10)); // 计算并缓存结果
console.log(fibonacci(10)); // 直接从缓存获取

16. 实现一个观察者模式（发布-订阅模式） ​

代码示例：

javascript
class EventEmitter {
  constructor() {
    this.events = new Map();
  }

  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(listener);
    return this;
  }

  emit(event, ...args) {
    if (this.events.has(event)) {
      this.events.get(event).forEach((listener) => listener(...args));
    }
    return this;
  }

  off(event, listener) {
    if (this.events.has(event)) {
      const listeners = this.events.get(event);
      const index = listeners.indexOf(listener);
      if (index !== -1) {
        listeners.splice(index, 1);
      }
    }
    return this;
  }

  once(event, listener) {
    const wrapper = (...args) => {
      listener(...args);
      this.off(event, wrapper);
    };
    this.on(event, wrapper);
    return this;
  }
}

// 使用示例
const emitter = new EventEmitter();

// 订阅事件
emitter.on("message", (data) => {
  console.log("Received message:", data);
});

// 发布事件
emitter.emit("message", "Hello, world!");

// 取消订阅
const callback = () => console.log("Special event");
emitter.on("special", callback);
emitter.off("special", callback);

javascript
function smoothScroll(targetId, duration) {
  const target = document.getElementById(targetId);
  if (!target) return;

  const startPosition = window.pageYOffset;
  const targetPosition = target.getBoundingClientRect().top;
  const startTime = performance.now();

  function scrollStep(timestamp) {
    const currentTime = timestamp - startTime;
    const progress = Math.min(currentTime / duration, 1);
    const easeProgress = easeInOutCubic(progress);

    window.scrollTo(0, startPosition + targetPosition * easeProgress);

    if (progress < 1) {
      requestAnimationFrame(scrollStep);
    }
  }

  // 缓动函数
  function easeInOutCubic(t) {
    return t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;
  }

  requestAnimationFrame(scrollStep);
}

// 使用示例
smoothScroll("section2", 1000); // 1 秒内平滑滚动到 id 为 section2 的元素

function generateToken(userId) {
  const payload = { userId };
  const secret = "your-secret-key";

  return jwt.sign(payload, secret, options);

  req.userId = payload.userId;
  next();
}

46. 什么是 MVC、MVP 和 MVVM 模式？它们有什么区别？ ​

MVC（Model-View-Controller）：

  • 模型（Model）：数据和业务逻辑
  • 视图（View）：用户界面

  • 模型（Model）：数据和业务逻辑
  • 视图（View）：展示界面，向 Presenter 发送用户操作
  • 展示器（Presenter）：处理视图逻辑，与模型交互

MVVM（Model-View-ViewModel）：

  • 模型（Model）：数据和业务逻辑
  • 视图（View）：HTML 模板
  • 视图模型（ViewModel）：绑定视图和模型，实现双向数据绑定

区别：

  • MVC：控制器直接操作视图
  • MVP：Presenter 与视图通过接口交互
  • MVVM：使用数据绑定和依赖注入，减少手动 DOM 操作

74. 什么是 Web Components？如何创建自定义元素？ ​

Web Components：一套标准，用于创建可复用的自定义 HTML 元素。

核心技术：

  1. Custom Elements：定义新 HTML 标签
  2. Shadow DOM：封装样式和结构
  3. HTML Templates：可复用的 HTML 片段
  4. HTML Imports：导入 HTML 文件（已被 ES 模块替代）

示例：

javascript
// 创建自定义元素
class MyButton extends HTMLElement {
  constructor() {
    super();

    // 创建shadow DOM
    const shadow = this.attachShadow({ mode: "open" });

    // 创建内部HTML
    shadow.innerHTML = `
            <style>
                button {
                    background-color: blue;
                    color: white;
                    padding: 10px 20px;
                    border: none;
                    border-radius: 5px;
                }
            </style>
            <button><slot>Default Text</slot></button>
        `;

    // 添加事件监听器
    const button = shadow.querySelector("button");
    button.addEventListener("click", () => {
      this.dispatchEvent(new Event("custom-click"));
    });
  }

  // 监听属性变化
  static get observedAttributes() {
    return ["disabled"];
  }

  attributeChangedCallback(name, oldValue, newValue) {
    if (name === "disabled" && this.shadowRoot) {
      const button = this.shadowRoot.querySelector("button");
      if (newValue !== null) {
        button.disabled = true;
        button.classList.add("disabled");
      } else {
        button.disabled = false;
        button.classList.remove("disabled");
      }
    }
  }
}

// 注册自定义元素
customElements.define("my-button", MyButton);

79. 如何优化 JavaScript 执行性能？ ​

优化技巧：

  1. 减少全局变量：

    javascript
    // 不好
function calculate() {
  result = a + b; // 隐式创建全局变量
}

// 好
function calculate() {
  const result = a + b;
  return result;
}

  2. 避免频繁 DOM 操作：

    javascript
    // 不好
for (let i = 0; i < 1000; i++) {
  document.body.appendChild(document.createElement("div"));
}

// 好
const fragment = document.createDocumentFragment();
for (let i = 0; i < 1000; i++) {
  fragment.appendChild(document.createElement("div"));
}
document.body.appendChild(fragment);

  3. 使用事件委托：

    javascript
    // 不好
document.querySelectorAll("button").forEach((button) => {
  button.addEventListener("click", handleClick);
});

// 好
document.body.addEventListener("click", (event) => {
  if (event.target.tagName === "BUTTON") {
    handleClick(event);
  }
});

  4. 优化循环：

    javascript
    // 缓存数组长度
for (let i = 0, len = array.length; i < len; i++) {
  // 循环体
}

// 使用for...of替代传统for循环
for (const item of array) {
  // 循环体
}

  5. 使用 requestAnimationFrame：

    javascript
    function animate() {
  requestAnimationFrame(() => {
    // 更新动画状态
    element.style.transform = `translateX(${position}px)`;
    if (position < target) animate();
  });
}
